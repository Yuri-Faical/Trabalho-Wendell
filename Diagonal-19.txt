#include <iostream>

using namespace std;

int main()
{
    int matriz[3][3];
    int soma = 0;

    cout << "Preencha a matriz 3x3 para calcular a diagonal principal:" << endl;
    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            cout << "Matriz[" << i << "][" << j << "]: ";
            cin >> matriz[i][j];
        }
    }

    // Somando apenas a diagonal principal (onde i == j)
    for (int i = 0; i < 3; i++)
    {
        soma += matriz[i][i];
    }

    cout << "\nMatriz:" << endl;
    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            cout << matriz[i][j] << "\t";
        }
        cout << endl;
    }

    cout << "\nA soma dos elementos da diagonal principal é: " << soma << endl;

    return 0;
}
