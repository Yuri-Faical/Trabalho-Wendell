#include <iostream>

using namespace std;

int main() {
    int vetor[10];
    bool ordenado = true;

    cout << "Digite 10 números para verificar se estão em ordem crescente:" << endl;
    for(int i = 0; i < 10; i++) {
        cout << "Valor " << i + 1 << ": ";
        cin >> vetor[i];
    }

    // Verifica se cada elemento é menor ou igual ao próximo
    for(int i = 0; i < 9; i++) {
        if(vetor[i] > vetor[i+1]) {
            ordenado = false; // Encontrou um elemento fora de ordem
            break;
        }
    }

    if(ordenado) {
        cout << "\nO vetor ESTÁ em ordem crescente." << endl;
    } else {
        cout << "\nO vetor NÃO está em ordem crescente." << endl;
    }

    return 0;
}
